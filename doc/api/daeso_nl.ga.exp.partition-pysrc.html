185      """ 
186      Greedy binning of objects of different sizes into n bins of approximately 
187      equal size (aka the partitioning problem).  
188       
189      @param objects: sequence of object 
190       
191      @param size: sequence of object sizes 
192   
193      @param n: number of bins 
194       
195      @return: a tuple of which the first element is a list of bin sizes and the 
196      second elements a list of bins, where each bin is in itself a list of 
197      objects. 
198      """ 
199      pairs = zip(sizes, objects) 
200      pairs.sort() 
201       
202      bins = [ [0, []] for i in range(n) ] 
203       
204      while pairs: 
205          size, obj = pairs.pop() 
206          smallest = bins[0] 
207          smallest[0] += size 
208          smallest[1].append(obj) 
209          bins.sort() 
210           
211      bin_sizes, bins = zip(*bins) 
212      return list(bin_sizes), list(bins)
