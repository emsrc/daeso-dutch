73 -def remove_stopwords(tokens, ignore_case=False, remove_punc=False, others=[]): 

74      """ 
75      Removes stopwords (by default only lower case)from a list of tokens, where 
76      ignore_case is a boolean indicating whether upper/lower case should be ignored,  
77      remove_punc is a boolean indicating whether punctuation is to be removed, and  
78      others is a list of additional stopwords. 
79      Returns filtered list of tokens. 
80      """ 
81      if remove_punc: 
82          remove = stopwords + list(punctuation) + others 
83      else: 
84          remove = stopwords + others 
85           
86      if ignore_case: 
87          return [t for t in tokens if t.lower() not in remove] 
88      else: 
89          return [t for t in tokens if t not in remove]
